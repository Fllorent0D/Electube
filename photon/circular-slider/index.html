Circular Slider

JavaScript API

Get Value

            var circSlider = document.getElementsByClassName("circular-slider")[0];

circSlider.value; // Returns an angle (0 - 400)

Set Value

            var circSlider = document.getElementsByClassName("circular-slider")[0];

var val = 200;
circSlider.setValue(val); // Angle from 0 - 400

Event 'change'

            var circSlider = document.getElementsByClassName("circular-slider")[0];

circSlider.addEventListener("change", function(event) {
  return event; // Change Event
});
